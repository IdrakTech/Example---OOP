	void addTask(Task task)
	{
		Task* temp = new Task[taskCount + 1];
		for (int i = 0; i < taskCount; i++)
		{
			temp[i] = tasks[i];
		}
		delete[] tasks;
		tasks = temp;
		tasks[taskCount] = task;
		taskCount++;
	}
